Setting up The Project

This project uses Docker for local development.

docker compose build
docker compose up -d
make setup_db

You can check the logs by running docker compose logs or docker compose logs -f to attach to the logs and follow the output.

You should now be able to access the app on your local machine by going to http://localhost:3000.

Dockerfile

FROM ruby:3.1

RUN apt-get update -qq && apt-get install -y nodejs postgresql-client
RUN mkdir /app

WORKDIR /app

COPY Gemfile* /app/

RUN gem install bundler && bundle install --with=development

COPY . /app

COPY entrypoint.sh /usr/bin/
RUN chmod +x /usr/bin/entrypoint.sh
ENTRYPOINT ["entrypoint.sh"]
EXPOSE 3000

# Start the main process.
CMD ["rails", "server", "-b", "0.0.0.0"]

Makefile

.PHONY: test seed

test:
	docker compose run --rm web bundle exec rspec

seed:
	docker compose run --rm web bundle exec rake db:seed

setup_db:
	docker compose run --rm web ./bin/rake db:setup
	docker compose run --rm web ./bin/rake db:migrate
	docker compose run --rm web ./bin/rake db:seed

migrate:
	docker compose run --rm web ./bin/rake db:migrate
